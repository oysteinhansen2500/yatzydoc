This class (component) renders a register page for the user, and handles hashing and sending the data to the server
onSubmit = (event) => {
  event.preventDefault();
  var salt = bcrypt.genSaltSync(10);


  this.setState({
    salt: salt,
    password: bcrypt.hashSync(this.state.password, salt)
  }, function() {
    fetch('http://87.248.16.163:15004/api/register', {
      method: 'POST',
      body: JSON.stringify(this.state),
      headers: {
        'Content-Type': 'application/json',
        'Accept': 'application/json'
      }
    }).then((response) => response.json()).then((messages) => {

      this.props.history.push(messages.redirectUrl);
    });
    !uses bcrypt for hashing before it is sent to the server!
